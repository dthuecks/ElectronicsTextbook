                                node[Dflipflop falling](D1){}
    ([xshift=0.5cm] D1.pin 6)   node[Dflipflop falling, anchor=pin 1](D2){}
    ([xshift=0.5cm] D2.pin 6)   node[Dflipflop falling, anchor=pin 1](D3){}
    ([xshift=0.5cm] D3.pin 6)   node[Dflipflop falling, anchor=pin 1](D4){}

    % Internal connections

    (D3.pin 3)                  to[short, -*] (\tikztostart |- j1)     
    
    
    % Inputs
    (j1)                        to[short, *-] ++(-0.5,0)
                                coordinate(C)
                                node[left](){$C$}

                                node[left](){In}

    % Outputs

                                node[above](){$Q_2$}
    ($(D3.pin 6)!0.5!(D4.pin 1)$)
                                to[short,*-] (\tikztostart |- Q1)
                                node[above](){$Q_3$}
    (D4.pin 6)                  to[short] (\tikztostart |- Q1)
                                node[above](){$Q_4$}
