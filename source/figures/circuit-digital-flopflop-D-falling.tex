        flipflop def={t1=$D$, t3=EN, t6=$Q$, t4={\ctikztextnot{$Q$}},
    }}}

    % Placement of digital components
    \draw(0, 0)             coordinate(start)
                            node[Dlatch, align=center](D1){Latch\\ 1}
    ([xshift=3cm] start)    node[Dlatch, align=center](D2){Latch\\ 2}
    ([yshift=-2cm] start)       
                            node[not port](not2){}

    % Internal connections
    (D1.pin 6)              to[short] (D2.pin 1)
    (D1.pin 3)              to[short] (\tikztostart |- not2.in)

                            to[short] (D2.pin 3)

    % Inputs

                            node[left](){$D$}

                           node[left](){CLK}

    % Outputs
