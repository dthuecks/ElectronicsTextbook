\begin{circuitikz}
    \draw (0,0) node[dipchip, num pins=8, hide numbers, external pins width=0, circuitikz/chips/scale=1](C){};
    \foreach \pin in {1,...,4} \node[font=\tiny, left] at(C.bpin \pin) {\pin};
    \foreach \pin in {5,...,8} \node[font=\tiny, right] at(C.bpin \pin) {\pin};
    \node [left, xshift=-2mm, font=\tiny, align=center] at (C.pin 1) {$V_\text{out,A}$};
    \node [left, xshift=-2mm, font=\tiny, align=center] at (C.pin 2) {$V_{-,A}$};
    \node [left, xshift=-2mm, font=\tiny, align=center] at (C.pin 3) {$V_{+,A}$};
    \node [left, xshift=-2mm, font=\tiny, align=center] at (C.pin 4) {GND};
    \node [right, xshift=+2mm, font=\tiny, align=center] at (C.pin 5) {$V_{+,B}$};
    \node [right, xshift=+2mm, font=\tiny, align=center] at (C.pin 6) {$V_{-,B}$};
    \node [right, xshift=+2mm, font=\tiny, align=center] at (C.pin 7) {$V_\text{out,B}$};
    \node [right, xshift=+2mm, font=\tiny, align=center] at (C.pin 8) {$V_{s+}$};
    % Place op-amp schematic connections inside chip pin diagram
    \path ($(C.pin 1)!0.4!(C.pin 3)$) coordinate (Avert);
    \path ($(C.pin 2)!0.3!(C.pin 7)$) coordinate (Ahoriz);
    \path ($(C.pin 2)!0.7!(C.pin 7)$) coordinate (Bhoriz)
            (Ahoriz |- Avert) node[op amp, scale=0.2, rotate=90](opampA){}
            (Bhoriz |- Avert) node[op amp, scale=0.2, xscale=-1, rotate=90](opampB){};
    
    % Connect pins
    \draw (C.pin 1) -| (opampA.out);
    \draw (C.pin 2) -- ($(C.pin 2)!0.1!(C.pin 7)$)
                    |- (opampA.-);
    \draw (C.pin 3) -| (opampA.+);
    \draw (C.pin 5) -| (opampB.+);
    \draw (C.pin 6) -| (opampB.-);
    \draw (C.pin 7) -- ($(C.pin 7)!0.1!(C.pin 2)$)
                    |- (opampB.out);
    \draw (C.pin 8) -- ($(C.pin 8)!0.5!(C.pin 1)$)
                    |- (opampA.down)
                    -- (opampB.down);
\end{circuitikz}
